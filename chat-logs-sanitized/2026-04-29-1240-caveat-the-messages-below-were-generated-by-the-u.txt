      243 +  /// 解析 VTT 时间行 "HH:MM:SS.mmm --> HH:MM:SS.mmm"
      244 +  /// 可能包含尾部设置如 "A:start"
      245 +  fn parse_vtt_time_line(line: &str) -> Option<(i64, i64)> {
      246 +      let line = line.trim();
      247 +      let arrow = line.find("-->")?;
      248 +      let start_str = line[..arrow].trim();
      249 +      let end_part = line[arrow + 3..].trim();
      250 +      // 去掉尾部设置（空格后的部分）
      251 +      let end_str = 
          + end_part.split_whitespace().next().unwrap_or(end_part);
      252 +  
      253 +      Some((parse_vtt_time(start_str)?, 
          + parse_vtt_time(end_str)?))
      254 +  }

      272 +          SubtitleFormat::Vtt => parse_vtt(&content),
      273 +      }
      274 +  }
